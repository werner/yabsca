def self.tree(id_node)
  if id_node.match(%rsrc:strats/) or id_node.match(%rsrc:root+\d+/)
    id = id_node.sub(%rsrc:strats/,"").to_i
    id = id_node.sub(%rsrc:root/,"").to_i if id == 0
    where(:strategy_id => id).collect { |perspective|
      node perspective
    }
  else
    []
  end
end
